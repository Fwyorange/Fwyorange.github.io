Python WebSpider Code

以下为IDLE的代码，为爬取网络小说的内容

from bs4 import BeautifulSoup
import requests

url=’爬取的网址’

req=requests.get(url)
req.encoding=’gbk’
bs=BeautifulSoup(req.text,’lxml’)
texts=bs.find_all(‘div’,class_=’showtxt’)

f=open(‘一念永恒.txt’,’w’)
f.write(texts[0].text.replace(‘\xa0’*8,’\n\n’))
f.close
